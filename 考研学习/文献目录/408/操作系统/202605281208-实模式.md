- 汇编中的`.xxx`, 是编译层面的, 并不是`cpu`指令层面的, `cpu`并不会执行
~~~c
#include "os.h"
.section .boot, "ax"  // 自定义boot段
.code16               // 16位
.global _start        // 全局入口

_start:
  cli                 // 禁止中断

  mov $0, %ax         // 实模式下段寄存器赋初值
  mov %ax, %ds
  mov %ax, %es
  mov %ax, %ss
  mov %ax, %fs
  mov %ax, %gs
  mov $_start, %esp  // 默认栈空间 

  sti                  // 开启中断

read_stage2:           // 第二阶段, 加载磁盘, 以及进入保护模式
  mov $_start_32, %bx  // 告诉cpu磁盘加载到内存的哪个地方, 也就是_start_32处
  mov $0x0002, %cx     // 从第几个扇区开始(2)
  mov $0x0240, %ax     // 2启动读磁盘功能, 40表示读取64个扇区(16进制的40是64)
  mov $0x0080, %dx     // 表示第一块硬盘
  int $0x0013          // 上述相当于传参, 这里是执行中断
  // 中断函数在ROM某处, 这里的地址是内存中实际地址(或者下标, 运算后会是地址)

  jc read_stage2       // 如果加载失败, 重新载入

hang:                  // ai给的, 我也不知道什么意思, 放过
  jmp hang
.org 0x01fe            // 当前段内偏移不满的地方填0
.byte 0x55, 0xaa       // 后续两位填0x55, 0xaa, 其实也是bootloader标志位

.section .text32, "ax" // 自定义32位段
.code32
.global _start_32

_start_32:
  jmp .
~~~

- `0x7c00`到低地址处, 默认是栈空间
- `0x7c00`到高地址`512`字节处, 是`bootloader`, 后两个字节通过判断是否为`0x55`与`0xaa`判断用户是否编写自己的`bootloader`
- `bootloader`放于第一扇区, `bois`自动加载到内存`0x7c00`处
- 这么小的空间肯定放不下所有的操作系统代码, 所以`bootloader`需要做的是加载扇区中操作系统的代码到内存中(也就是`read_stage2`所做的)

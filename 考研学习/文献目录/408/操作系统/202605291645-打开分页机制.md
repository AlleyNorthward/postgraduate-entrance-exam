~~~C
#include "os.h"

typedef unsigned char u8;
typedef unsigned short u16;
typedef unsigned int u32;

#define PDE_P   (1 << 0) // 存在位
#define PDE_W   (1 << 1) // 读写位
#define PDE_U   (1 << 2) // 权限位
#define PDE_PS  (1 << 7) // 开启4M读取

u32 pg_dir[1024] __attribute__((aligned(4096))) = {
  [0] = (0) | PDE_P | PDE_W | PDE_U | PDE_PS
};
~~~
- 这个是**页目录表**, `4KB`读取, 还会有个二级目录表, `4M`读取, 只有这一级目录表
- 共`1024`表项
- **基地址**`4096`对齐, 也就是`4kb`对齐, 强制要求, 因为该地址会放入`cr3`寄存器中, 硬件强制要求
- 内部是具体表项配值
~~~C
#include "os.h"
.section .boot, "ax"
.code16
.global _start

_start:
  cli

  mov $0, %ax
  mov %ax, %ds
  mov %ax, %es
  mov %ax, %ss
  mov %ax, %fs
  mov %ax, %gs
  mov $_start, %esp

  sti

read_stage2:
  mov $_start_32, %bx
  mov $0x0002, %cx
  mov $0x0240, %ax
  mov $0x0080, %dx
  int $0x0013

  jc read_stage2

  cli
  lgdt [gdt_desc]

  mov $1, %eax
  lmsw %ax
  jmp $KERNEL_CODE_SEG, $_start_32

hang:
  jmp hang

.org 0x01fe
.byte 0x55, 0xaa

.section .text32, "ax"
.code32
.global _start_32

_start_32:
  mov $KERNEL_DATA_SEG, %ax
  mov %ax, %ds
  mov %ax, %es
  mov %ax, %ss
  mov %ax, %gs
  mov %ax, %fs
  mov $_start, %esp

  mov $pg_dir, %eax
  mov %eax, %cr3

  mov %cr4, %eax             # 开启4M页目录表
  orl $(1 << 4), %eax
  mov %eax, %cr4

  mov %cr0, %eax             # 打开分页机制
  orl $(1 << 31), %eax
  mov %eax, %cr0
  jmp .

.section .data
.align 8

gdt_table:
  .quad 0x0000000000000000
  .quad 0x00cf9a000000ffff
  .quad 0x00cf92000000ffff
gdt_end:

gdt_desc:
.word gdt_end - gdt_table - 1
.long gdt_table
~~~

- `87-97`是打开分页机制的代码了
- 起始就是配置目录表, 开启相关标志位
- 开启后, `monitor info mem`就可以看到相关映射情况了. 这里没下, 所以是全0映射.

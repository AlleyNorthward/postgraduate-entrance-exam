# 源代码测试
~~~C
#include "os.h"
.section .boot, "ax"
.code16
.global _start

_start:
  cli

  mov $0, %ax
  mov %ax, %ds
  mov %ax, %es
  mov %ax, %ss
  mov %ax, %fs
  mov %ax, %gs
  mov $_start, %esp

  sti

read_stage2:
  mov $_start_32, %bx
  mov $0x0002, %cx
  mov $0x0240, %ax
  mov $0x0080, %dx
  int $0x0013

  jc read_stage2

  cli
  lgdt [gdt_desc]                   // 将C数组跟lgdt建立联系, 方便后续查找

  mov $1, %eax
  lmsw %ax                          // cr0位 置1
  jmp $KERNEL_CODE_SEG, $_start_32  // 根据段描述符寻址执行

hang:
  jmp hang
.org 0x01fe
.byte 0x55, 0xaa

.section .text32, "ax"
.code32
.global _start_32

_start_32:
  mov $KERNEL_DATA_SEG, %ax          // 32位保护模式下配置相关寄存器
  mov %ax, %ds
  mov %ax, %es
  mov %ax, %ss
  mov %ax, %gs
  mov %ax, %fs
  mov $_start, %esp

  mov $0x12345678, %eax              // 栈push, pop 测试
  push %eax

  mov $0, %ebx
  pop %ebx
  jmp .

gdt_desc:                            // 这里相当于结构体
.word (256 * 8) - 1
.long gdt_table
~~~

~~~C
#ifndef OS_H
#define OS_H

#define KERNEL_CODE_SEG 8
#define KERNEL_DATA_SEG 16

#endif // OS_H

#include "os.h"

typedef unsigned char u8;
typedef unsigned short u16;
typedef unsigned int u32;

struct {
  u16 limit_l;
  u16 base_l;
  u16 basehl_attr;
  u16 base_limit;
} gdt_table[256] __attribute__((aligned(8))) = {
    [KERNEL_CODE_SEG / 8] = {0xffff, 0x0000, 0x9a00, 0x00cf},
    [KERNEL_DATA_SEG / 8] = {0xffff, 0x0000, 0x9200, 0x00cf}};
~~~
- 后续我们可以调试看看, 实际上发生了什么.这里需要用到[[202605282035-大小端|大小端]]可以了解下
- 可以看看[[202605291018-GDTR寄存器|GDTR寄存器]]的描述
- 此时刚给lgdt赋完了值
![[Pasted image 20260528204941.png]]
- 我们看看gdt的值是什么
![[Pasted image 20260528205033.png]]
- 那么什么含义呢? 此时需要关键表项, 也就是`gdt_desc`以及`gdt_table`了, 我们先看前者地址
![[Pasted image 20260528205142.png]]
- 看其值
![[Pasted image 20260528205218.png]]
- `lgdt`会读取其六字节 
- 从上可见,六字节是`ff0700c00000`, 变成值的话是`0000c000  07ff`
- 然后呢, 后面两位是大小, 前面四位是基地址. 中间补了0, 所以没有问题, 逻辑行的通
- 再看`gdt_table`
![[Pasted image 20260528211920.png]]
- 很巧哦, `gdt_table`的值, 正好是`gdt`寄存器的基地址, 那么`07ff`是什么含义呢? 
	- 确实是偏移, 但实际上是数组大小
	- `0x07ff` + `1` = `0x8000` = `2048`字节
	- 因为我赋值是`256`, 也就是可以有`256`个描述符, 总共`2048`字节
		- 在非裸机编程中, 我们更多地会用`sizeof`, `malloc`时, 数组总字节数是通过`sizeof * size`获取的, 实际上就是总字节数
- `jmp $KERNEL_CODE_SEG, $_start_32` 进入保护模式后, 就这么寻址执行了. 前者是描述符, `cpu`会去`gdt`寄存器找相关说明. `gdt`寄存器存的什么? 上面说了, 存的是基地址+大小. 执行完这行后, `KERNEL_CODE_SEG`的值会放入`CS`寄存器中, 如下
![[Pasted image 20260528213937.png]]
- 所以为什么`KERNEL_CODE_SEG`的值是`8`呢? 因为它就是从第三位(`base-1`)开始的, 所以虽然是`8`, 实际上就是数组的下标`1`, 因为代码中存储`gdt_table`时进行了除`8`操作(左移`3`位)
- 此时`cs`位可以认为是数组下标, 而`gdt`中有数组基地址, 此时就能找到对应的`gdt`表, `gdt`表中有更多详细信息, 例如虚拟地址空间中的首地址. 这里默认的是`0x00`, 所以最终还是会前往`_start_32`执行对应代码.
- 我们再回想一下`cpu`寻址的逻辑. 
	- 保护模式下寻址, 会配置段寄存器
	- 例如`jmp $KERNEL_CODE_SEG ,$_start_32`, `KERNEL_CODE_SEG`值会存入`CS`寄存器中
	- 此时的`gdt`寄存器中已经存有`gdt`表的首地址和大小了
	- `cpu`会根据`cs`寄存器的段选择子(偏移), 找到表的基地址, 根据选择子(偏移)找到实际表项
	- 根据实际表项访问对应的虚拟内存空间

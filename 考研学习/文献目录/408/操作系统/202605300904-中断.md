- 中断的发生, 实际上就是执行函数去了, 那么实际上只需要存储选择子+偏移即可, 实际上确实是这么做的
~~~C
#include "os.h"

typedef unsigned char u8;
typedef unsigned short u16;
typedef unsigned int u32;

#define MAP_ADDR 0x80000000

extern u32 pg_dir[1024];
extern void timer_init(void);

u8 map_phy_buffer[4096] __attribute__((aligned(4096))) = {0x35};
static u32 page_table[1024] __attribute__((aligned(4096))) = {1};

struct {
  u16 offset_l;
  u16 selector;
  u16 attr;
  u16 offset_h;
} idt_table[256] __attribute__((aligned(8))) = {1};

void outb(u8 data, u16 port) {
  __asm__ __volatile__("outb %[v], %[p]"::[p]"d"(port), [v]"a"(data));
}

void os_init(void) {
  outb(0x11, 0x20);
  outb(0x11, 0xa0);
  outb(0x20, 0x21);
  outb(0x28, 0xa1);
  outb((1 << 2), 0x21);
  outb(2, 0xa1);
  outb(0x1, 0x21);
  outb(0x1, 0xa1);
  outb(0xfe, 0x21);
  outb(0xff, 0xa1);

  int tmo = 1193180 / 100;
  outb(0x36, 0x43);
  outb((u8)tmo, 0x40);
  outb(tmo >> 8, 0x40);

  idt_table[0x20].offset_l = (u32)timer_init & 0xffff;
  idt_table[0x20].offset_h = (u32)timer_init >> 16;
  idt_table[0x20].selector = KERNEL_CODE_SEG;
  idt_table[0x20].attr = 0x8e00;

  pg_dir[MAP_ADDR >> 22] = (u32)page_table | PDE_P | PDE_W | PDE_U;
  page_table[(MAP_ADDR >> 12) & 0x3FF] =
      (u32)map_phy_buffer | PDE_P | PDE_W | PDE_U;
}
~~~

~~~C
#include "os.h"
.section .boot, "ax"
.code16
.global _start

_start:
  cli

  mov $0, %ax
  mov %ax, %ds
  mov %ax, %es
  mov %ax, %ss
  mov %ax, %fs
  mov %ax, %gs
  mov $_start, %esp

  sti

read_stage2:
  mov $_start_32, %bx
  mov $0x0002, %cx
  mov $0x0240, %ax
  mov $0x0080, %dx
  int $0x0013

  jc read_stage2

  cli
  lgdt [gdt_desc]
  lidt [idt_desc]

  mov $1, %eax
  lmsw %ax
  jmp $KERNEL_CODE_SEG, $_start_32

hang:
  jmp hang

.org 0x01fe
.byte 0x55, 0xaa

.section .text32, "ax"
.code32
.global _start_32, timer_init

_start_32:
  mov $KERNEL_DATA_SEG, %ax
  mov %ax, %ds
  mov %ax, %es
  mov %ax, %ss
  mov %ax, %gs
  mov %ax, %fs
  mov $_start, %esp

  call os_init

  mov $pg_dir, %eax
  mov %eax, %cr3

  mov %cr4, %eax
  orl $(1 << 4), %eax
  mov %eax, %cr4

  mov %cr0, %eax
  orl $(1 << 31), %eax
  mov %eax, %cr0

  sti
  jmp .

timer_init:
  push %ds
  pusha
  mov $0x20, %al
  outb %al, $0x20

  popa
  pop %ds
  iret

.section .data
.align 8

gdt_table:
  .quad 0x0000000000000000
  .quad 0x00cf9a000000ffff
  .quad 0x00cf92000000ffff
gdt_end:

gdt_desc:
.word gdt_end - gdt_table - 1
.long gdt_table

idt_desc:
.word (256 * 8) - 1
.long idt_table

.align 4096
.global pg_dir

pg_dir:
.long 0 + PDE_P + PDE_W + PDE_U + PDE_PS
.fill 1023, 4, 0
~~~
- 汇编中通过`lidt`来配置idt寄存器
- 所以我们思考一下, 中断发生时, 函数是如何执行的
- 定时器中断发生, `cpu`会去找到`idt`寄存器
- `idt`寄存器内部有表, 根据定时器中断的类型(下标), 可以找到具体的表项
- 根据具体表项, 可以确定描述符以及偏移
- 根据描述符外加`gdt`寄存器信息, 可以找到首地址
- 根据首地址外加偏移, 可以找到具体函数地址, 然后执行
- 如果开启分页机制的话, 就多了层映射, 但终归还是能够找到实际的物理地址的

# 注意
- 如果发生中断, 中断函数中如果使用某些寄存器, 那么当结束中断时, 原有的寄存器信息就消失了
- 所以需要手动保存函数栈, 也就是`timer_init`中所做的
- 所以`call`指令会自动保存信息
- 而中断实际上并不是函数, 但是可以模拟出函数的效果

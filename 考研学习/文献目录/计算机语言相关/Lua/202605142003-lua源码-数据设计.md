#### 自己设计动态类型, 要求是类型分离.
~~~C
#define LUA_CFUNCTION 0
typedef struct lua_State lua_State;
typedef int (*lua_CFunction)(lua_State *L);
typedef long lua_integer;
typedef double lua_number;

typedef union lua_Value {
  lua_CFunction f;
  lua_integer i;
  lua_number n;
  int b;
} Value;

typedef struct lua_TValue {
  Value v_;
  int tt_;
} TValue;

int f(lua_State *L);
int main() {
  TValue v;
  v.v_.f = f;
  v.tt_ = LUA_CFUNCTION;
}
~~~
- 参考代码, 绝不是非得要记住复刻的!

#### 请写一下跟数据相关的set方法
~~~C
typedef struct lua_State lua_State;
typedef int (*lua_CFunction)(lua_State *L);
typedef long lua_Integer;
typedef double lua_Number;
typedef union lua_Value {
  int b;
  lua_CFunction f;
  lua_Integer i;
  lua_Number n;
} Value;

typedef struct lua_TValue {
  Value value_;
  int tt_;
} TValue;

typedef TValue *StkId;

void setbvalue(StkId target, int bo) {
  target->value_.b = bo;
  target->tt_ = 0;
}

void setcfunction(StkId target, lua_CFunction f) {
  target->value_.f = f;
  target->tt_ = 1;
}

void setivalue(StkId target, lua_Integer i) {
  target->value_.i = i;
  target->tt_ = 2;
}

void setnvalue(StkId target, lua_Number n) {
  target->value_.n = n;
  target->tt_ = 3;
}
~~~

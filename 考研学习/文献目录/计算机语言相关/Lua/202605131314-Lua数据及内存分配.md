#include <assert.h>
#include <stddef.h>
#include <stdlib.h>

#define LUA_TNIL 5
#define LUA_TFUNCTION 7
#define LUA_TNUMBER 1
#define LUA_TTHREAD 8
#define LUA_TLCF (LUA_TFUNCTION | (1 << 4))
#define LUA_NUMINT (LUA_TNUMBER | (0 << 4))

typedef struct lua_State lua_State;
typedef long lua_Integer;
typedef unsigned char lu_byte;
typedef int (*lua_CFunction)(lua_State *L);
typedef void *(*lua_Alloc)(void *ptr, size_t osize, size_t nsize);

typedef union lua_Value {
  lua_Integer i;
  lua_CFunction f;
} Value;

typedef struct lua_TValue {
  Value value_;
  int tt_;
} TValue;

typedef TValue *StkId;

void setivalue(StkId target, int integer) {
  target->value_.i = integer;
  target->tt_ = LUA_NUMINT;
}

void setfvalue(StkId target, lua_CFunction f) {
  target->value_.f = f;
  target->tt_ = LUA_TLCF;
}

void setnilvalue(StkId target) { target->tt_ = LUA_TNIL; }

typedef struct global_State global_State;

typedef struct lua_State {
  StkId stack;
  StkId stack_last;
  StkId top;
  int stack_size;
  global_State *l_G;
} lua_State;

typedef struct global_State {
  lua_State *mainthread;
  lua_Alloc frealloc;
  lua_CFunction panic;
} global_State;

void *luaM_realloc(lua_State *L, void *ptr, size_t osize, size_t nsize) {
  global_State *g = L->l_G;
  int oldsize = ptr ? osize : 0;
  void *ret = g->frealloc(ptr, oldsize, nsize);
  return ret;
}

static void stack_init(lua_State *L) {
  L->stack = (StkId)luaM_realloc(L, NULL, 0, 40 * sizeof(TValue));
  L->stack_size = 40;
  L->stack_last = L->stack + 40;
  L->top = L->stack;

  for (int i = 0; i < L->stack_size; ++i) {
    setnilvalue(L->stack + i);
  }
  ++L->top;
}

static void *l_alloc(void *ptr, size_t osize, size_t nsize) {
  if (nsize == 0) {
    free(ptr);
    return NULL;
  }
  return realloc(ptr, nsize);
}

#define LUA_EXTRASPACE sizeof(void *)

typedef struct LX {
  lu_byte extra_[LUA_EXTRASPACE];
  lua_State l;
} LX;

typedef struct LG {
  LX l;
  global_State g;
} LG;

lua_State *lua_newstate(lua_Alloc alloc) {
  global_State *g;
  lua_State *L;

  LG *lg = (LG *)alloc(NULL, LUA_TTHREAD, sizeof(LG));
  if (!lg) {
    return NULL;
  }

  g = &lg->g;
  g->frealloc = alloc;
  L = &lg->l.l;
  L->l_G = g;
  g->mainthread = L;
  stack_init(L);

  return L;
}

void increase_top(lua_State *L) {
  ++L->top;
  assert(L->top <= L->stack_last);
}

void lua_pushcfunction(lua_State *L, lua_CFunction f) {
  setfvalue(L->top, f);
  increase_top(L);
}

void lua_pushinteger(lua_State *L, int integer) {
  setivalue(L->top, integer);
  increase_top(L);
}

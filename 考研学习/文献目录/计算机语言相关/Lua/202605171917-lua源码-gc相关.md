- 下面代码没有任何保护, 存在内存泄漏的风险, 知道就行
~~~C
#include <stdbool.h>
#include <stdio.h>
#include <stdlib.h>

#define MAX_STACK 10

typedef struct Object {
  int id;
  bool marked;
  struct Object *next;
  struct Object **references;
  int ref_count;
} Object;

Object *allgc = NULL;

typedef struct Stack {
  Object **base;
  Object **top;
  int size;
} Stack;

Object *new_object(int id) {
  Object *obj = (Object *)malloc(sizeof(Object));
  if (!obj)
    return NULL;

  obj->id = id;
  obj->marked = false;
  obj->next = allgc;
  obj->references = NULL;
  obj->ref_count = 0;

  allgc = obj;

  printf("创建对象: obj%d\n", obj->id);
  return obj;
}

Stack *stack_init() {
  Stack *stack = (Stack *)malloc(sizeof(Stack));
  stack->base = (Object **)malloc(sizeof(Object *) * MAX_STACK);

  stack->size = MAX_STACK;
  stack->top = stack->base;

  return stack;
}

void push(Stack *stack, Object *obj) {
  if (stack->top >= stack->base + MAX_STACK - 1) {
    printf("栈溢出!\n");
    return;
  }
  *(stack->top++) = obj;
}

Object *pop(Stack *stack) {
  if (stack->top <= stack->base) {
    printf("栈已空!\n");
    return NULL;
  }
  return *(--stack->top);
}

void add_reference(Object *from, Object *to) {
  Object **new_refs = (Object **)realloc(
      from->references, sizeof(Object *) * (from->ref_count + 1));

  if (!new_refs) {
    printf("引用分配失败!\n");
    return;
  }

  from->references = new_refs;
  from->references[from->ref_count++] = to;
}

void print_all_objects(Stack *stack) {
  printf("\n==== 当前 GC 堆 ====\n");

  Object *p = allgc;
  while (p) {
    printf("obj%d%s -> ", p->id, p->marked ? "(M)" : "");
    p = p->next;
  }
  printf("NULL\n");

  printf("==== 栈 ====\n");
  for (Object **cur = stack->base; cur < stack->top; cur++) {
    printf("obj%d ", (*cur)->id);
  }
  printf("\n");
}

void mark(Object *obj) {
  if (!obj || obj->marked)
    return;

  obj->marked = true;
  printf("标记 obj%d\n", obj->id);

  for (int i = 0; i < obj->ref_count; i++) {
    mark(obj->references[i]);
  }
}

void mark_phase(Stack *stack) {
  printf("\n=== MARK 阶段 ===\n");

  for (Object **ptr = stack->base; ptr < stack->top; ptr++) {
    mark(*ptr);
  }
}

void sweep_phase() {
  printf("\n=== SWEEP 阶段 ===\n");

  while (allgc) {
    Object *obj = allgc;

    if (!obj->marked) {
      printf("释放 obj%d\n", obj->id);
      allgc = allgc->next;

      free(obj->references);
      free(obj);
    } else {
      obj->marked = false;
      allgc = allgc->next;
    }
  }
}

void _sweep_phase() {
  printf("\n=== SWEEP 阶段 ===\n");
  Object **ptr = &allgc;
  while (*ptr) {
    Object *obj = *ptr;

    if (!obj->marked) {
      printf("释放 obj%d\n", obj->id);
      *ptr = obj->next;
      free(obj->references);
      free(obj);
    } else {
      obj->marked = false;
      ptr = &obj->next;
    }
  }
}

void gc(Stack *stack) {
  printf("\n================ GC START ================\n");

  mark_phase(stack);
  _sweep_phase();

  printf("=============== GC END ===================\n");
}

void sweep_all(Stack *stack) {
  printf("\n=== FINAL CLEANUP ===\n");

  Object *p = allgc;
  while (p) {
    Object *next = p->next;
    free(p->references);
    free(p);
    p = next;
  }
  allgc = NULL;

  free(stack->base);
  free(stack);

  printf("全部资源释放完成\n");
}

int main() {
  printf("===== Mark-Sweep GC Demo =====\n\n");

  Object *obj1 = new_object(1);
  Object *obj2 = new_object(2);
  Object *obj3 = new_object(3);
  Object *obj4 = new_object(4);
  Object *obj5 = new_object(5);
  Object *obj6 = new_object(6);
  Object *obj7 = new_object(7);

  Stack *stack = stack_init();

  add_reference(obj1, obj3);
  add_reference(obj3, obj7);
  add_reference(obj5, obj6);

  push(stack, obj1);
  push(stack, obj2);
  push(stack, obj5);
  push(stack, obj4);

  print_all_objects(stack);

  pop(stack);
  pop(stack);

  print_all_objects(stack);

  gc(stack);

  print_all_objects(stack);

  sweep_all(stack);

  printf("\n演示结束\n");
  return 0;
}
~~~

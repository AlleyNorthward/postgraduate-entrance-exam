#### lua的异常设计是通过链表, 来传递函数栈信息的. 这里请模拟一下dfs链式链接的代码吧. 提示, 需要有lua_longjmp结构体和lua_State结构体, 根据需要选择对应属性. 代码逻辑自己设计.
~~~C
#include <setjmp.h>
#include <stdio.h>

jmp_buf b;

#define TRY if (setjmp(b) == 0)
#define CATCH else
#define THROW(b) longjmp(b, 1);

typedef struct lua_longjmp {
  struct lua_longjmp *previous;
  int status;
} lua_longjmp;

typedef struct lua_State {
  lua_longjmp *errorjmp;
} lua_State;

void dfs(lua_State *L, int count) {
  if (count == 5)
    return;
  lua_longjmp lj;
  lj.previous = L->errorjmp;
  lj.status = count;
  printf("%d\n", lj.status);

  L->errorjmp = &lj;
  dfs(L, ++count);

  L->errorjmp = lj.previous;
}

int main() {
  lua_State l;
  l.errorjmp = NULL;
  dfs(&l, 0);
  return 0;
}
~~~

#### 在pcall执行时, 会给每个函数生成函数信息, lua官方取名叫做next_ci(), 内部跟链表有一定联系当然跟[[202605142042-lua源码-内存分配|内存分配]], [[202605161443-lua源码-C函数调用|C函数调用]]也是有联系的, 我打算放在这里了. 请实现一下该函数. 提示CallInfo* next_ci(lua_State* L, StkId func, int nresuslt)
~~~C
CallInfo* next_ci(lua_State* L, StkId func, int nresult){
  global_State* g = L->G_l;
  CallInfo* ci = (CallInfo*)g->frealloc(g->ud, NULL, 0, sizeof(CallInfo));
  ci->func = func;
  ci->top = func + LUA_MINSTACK;
  ci->nresult = nresult;
  ci->callstatus = LUA_OK;
  ci->previous = L->ci;
  ci->next = NULL;
  L->ci = ci;
}
~~~
- 整体思想是, **配置**新节点, **链接**, **更新**指针

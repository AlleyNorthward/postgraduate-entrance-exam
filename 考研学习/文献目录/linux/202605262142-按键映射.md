# 大写跟esc
~~~linux
sudo sed -i 's/^XKBOPTIONS=.*/XKBOPTIONS="caps:swapescape"/' /etc/default/keyboard
grep -q '^XKBOPTIONS=' /etc/default/keyboard || echo 'XKBOPTIONS="caps:swapescape"' | sudo tee -a /etc/default/keyboard
sudo setupcon
sudo systemctl restart keyboard-setup
~~~

# 恢复
~~~linux
sudo sed -i 's/^XKBOPTIONS=.*/XKBOPTIONS=""/' /etc/default/keyboard
sudo setupcon
sudo reboot
~~~

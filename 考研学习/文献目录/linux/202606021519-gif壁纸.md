#!/usr/bin/env python3
import warnings
warnings.filterwarnings("ignore", category=DeprecationWarning)

import gi
gi.require_version("Gtk", "3.0")
gi.require_version("Gdk", "3.0")
gi.require_version("GdkPixbuf", "2.0")
gi.require_foreign("cairo")

import cairo
from gi.repository import Gtk, Gdk, GdkPixbuf, GLib

GIF = "/home/alley/wallpaper/2717323779/nervous-dog-spinning.gif"

W = 128
H = 128
x = 80
y = 80
dx = 3
dy = 3
BOTTOM_MARGIN = 70

# True: 小狗在终端下面；False: 小狗在窗口上面
BELOW_NORMAL_WINDOWS = True

class Doge(Gtk.Window):
    def __init__(self):
        super().__init__(type=Gtk.WindowType.TOPLEVEL)

        self.set_title("doge-wallpaper")
        self.set_name("doge-wallpaper")
        self.set_decorated(False)
        self.set_resizable(False)
        self.set_app_paintable(True)
        self.set_skip_taskbar_hint(True)
        self.set_skip_pager_hint(True)
        self.set_accept_focus(False)
        self.set_focus_on_map(False)
        self.stick()

        if BELOW_NORMAL_WINDOWS:
            self.set_keep_below(True)
            self.set_type_hint(Gdk.WindowTypeHint.DESKTOP)
        else:
            self.set_keep_above(True)

        screen = self.get_screen()
        visual = screen.get_rgba_visual()
        if visual:
            self.set_visual(visual)

        self.anim = GdkPixbuf.PixbufAnimation.new_from_file(GIF)
        self.iter = self.anim.get_iter(None)

        self.resize(W, H)
        self.move(x, y)

        self.connect("draw", self.on_draw)

        GLib.timeout_add(40, self.next_frame)
        GLib.timeout_add(20, self.move_doge)

    def on_draw(self, widget, cr):
        cr.set_operator(cairo.OPERATOR_CLEAR)
        cr.paint()
        cr.set_operator(cairo.OPERATOR_OVER)

        pixbuf = self.iter.get_pixbuf()
        Gdk.cairo_set_source_pixbuf(cr, pixbuf, 0, 0)
        cr.paint()
        return False

    def next_frame(self):
        self.iter.advance(None)
        self.queue_draw()
        return True

    def move_doge(self):
        global x, y, dx, dy

        display = Gdk.Display.get_default()
        monitor = display.get_primary_monitor() or display.get_monitor(0)
        geo = monitor.get_geometry()

        sw = geo.width
        sh = geo.height

        x += dx
        y += dy

        if x <= 0 or x + W >= sw:
            dx = -dx

        if y <= 0 or y + H >= sh - BOTTOM_MARGIN:
            dy = -dy

        self.move(x, y)
        return True

win = Doge()
win.show_all()
Gtk.main()

chmod +x ~/.local/bin/doge-bounce

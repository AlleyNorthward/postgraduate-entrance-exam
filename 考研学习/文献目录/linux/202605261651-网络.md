# 临时静态ip设置
~~~linux
sudo ip link set enp2s0 up
sudo ip addr flush dev enp2s0
sudo ip addr add 192.168.137.2/24 dev enp2s0
sudo ip route add default via 192.168.137.1
echo "nameserver 8.8.8.8" | sudo tee /etc/resolv.conf
~~~
# 永久设置
~~~ymal
sudo nano /etc/netplan/00-installer-config.yaml
network:
  version: 2
  ethernets:
    enp2s0:
      dhcp4: no
      addresses:
        - 192.168.137.2/24
      routes:
        - to: default
          via: 192.168.137.1
      nameservers:
        addresses:
          - 8.8.8.8
          - 1.1.1.1
sudo netplan try
sudo netplan apply
ip addr show enp2s0
ip route
resolvectl status
~~~

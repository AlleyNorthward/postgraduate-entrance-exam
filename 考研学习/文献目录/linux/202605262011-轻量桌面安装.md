# openbox 启动

~~~linux
mkdir -p ~/.config/openbox

cat > ~/.config/openbox/autostart <<'EOF'
tint2 &
xfce4-terminal &
EOF

cat > ~/.xinitrc <<'EOF'
#!/bin/sh
xsetroot -solid "#2C001E"
exec openbox-session
EOF

chmod +x ~/.xinitrc
~~~

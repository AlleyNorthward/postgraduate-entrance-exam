# 安装
~~~linux
sudo apt update
sudo apt install -y openssh-server
~~~
# 启动
~~~linux
sudo systemctl enable ssh
sudo systemctl start ssh
~~~
# 检查
~~~linux
sudo systemctl status ssh
~~~
# 连接
~~~linux
	ssh alley@192.168.137.2
~~~

# 异网ssh连接方案
~~~C
sudo nano /etc/ssh/sshd_config
文件后添加:
AllowTcpForwarding yes
GatewayPorts yes

sudo systemctl restart ssh
sudo systemctl status ssh

B电脑执行
ssh -R 0.0.0.0:2222:localhost:22 root@115.28.208.100

云端验证:
ss -lntp | grep 2222

linux测试:
nc -vz 115.28.208.100 2222

连接
ssh -p 2222 HP@115.28.208.100
~~~

# 查看相关wifi模块版本信息
~~~linux
sudo lspci | grep -i network
~~~

~~~linux
sudo apt update && sudo apt upgrade -y
sudo apt install dkms linux-headers-$(uname -r) build-essential -y
sudo apt install broadcom-sta-dkms -y

sudo modprobe -r b43 bcma brcmsmac
echo "blacklist brcmsmac" | sudo tee /etc/modprobe.d/blacklist-broadcom.conf
echo "blacklist b43" | sudo tee -a /etc/modprobe.d/blacklist-broadcom.conf
echo "blacklist bcma" | sudo tee -a /etc/modprobe.d/blacklist-broadcom.conf
sudo update-initramfs -u

sudo modprobe wl
ip link show
~~~

# 连接wifi
~~~linux
sudo apt update
sudo apt install -y network-manager
sudo systemctl enable --now NetworkManager

# 扫描并列出所有WiFi信号
nmcli dev wifi list
nmcli dev wifi connect "你的WiFi名称" password "你的WiFi密码"
# 看网卡状态
nmcli device status

# 测试能不能上网
ping -c 3 www.baidu.com
~~~

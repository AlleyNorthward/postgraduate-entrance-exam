#### 官方提供的newstate()外部接口中, 并没有参数, 但是内部接口中实际上是有参数的, 提供了内存分配策略以及用户数据传递. 请编写一下lua_newstate()函数, 并且初始化global_state和lua_state结构体. 按照自己的思想写就行, 不用非得记住什么.  注意: 不用对栈初始化

~~~C
#include <stdlib.h>

void *l_alloc(void *ud, void *ptr, size_t osize, size_t nsize) {
  (void)ud;
  (void)osize;

  if (nsize == 0) {
    free(ptr);
    return NULL;
  }

  return realloc(ptr, nsize);
}

typedef struct lua_State lua_State;
typedef void *(*lua_Alloc)(void *ud, void *ptr, size_t osize, size_t nsize);
typedef struct global_State {
  lua_State *mainthread;
  void *ud;
  lua_Alloc al;
} global_State;

struct lua_State {
  global_State *g;
};

typedef struct LX {
  lua_State l;
} LX;

typedef struct LG {
  LX l;
  global_State g;
} LG;

lua_State* lua_newstate(lua_Alloc alloc, void *ud) {
  lua_State *L;
  global_State *g;
  LG *lg = (LG *)alloc(ud, NULL, 0, sizeof(LG));

  g = &lg->g;
  g->ud = ud;
  g->al = alloc;

  L = &lg->l.l;
  L->g = g;
  g->mainthread = L;

  return L;
}
~~~

#### 自己写一下栈初始化代码吧, 按照自己的思想来写就好
~~~C
#include <stdlib.h>
#define LUA_STACKSIZE 20
#define LUA_MINSIZE 5

typedef struct Value Value;
typedef struct TValue {

} TValue;

typedef TValue *StkId;
void setnilvalue(StkId target);
typedef void *(*lua_Alloc)(void *ud, void *ptr, size_t osize, size_t nsize);

typedef struct lua_State lua_State;
typedef struct CallInfo {
  StkId func;
  StkId top;
} CallInfo;

typedef struct global_State {
  lua_State *mainthread;
  lua_Alloc frealloc;
  void *ud;
} global_State;

struct lua_State {
  global_State *g;
  StkId stack;
  StkId top;
  StkId stack_last;
  int stack_size;
  CallInfo base_ci;
  CallInfo *ci;
};

void stack_init(lua_State *L) {
  global_State *g;
  L->stack = (StkId)g->frealloc(g->ud, NULL, 0, LUA_STACKSIZE * sizeof(TValue));

  L->top = L->stack;
  L->stack_size = LUA_STACKSIZE;
  L->stack_last = L->stack + LUA_STACKSIZE;

  for (int i = 0; i < LUA_STACKSIZE; ++i) {
    setnilvalue(L->stack + i);
  }
  ++L->top;

  L->ci = &L->base_ci;
  L->ci->func = L->stack;
  L->ci->top = L->stack + LUA_MINSIZE;
}
~~~

#### lua中有个函数是index2addr(). 分析完lua_settop()后, 这个函数也就变得十分简单明确了. 请自己编写实现一下
~~~C
TValue* index2addr(Lua_State* L, int idx){
  if(idx >= 0){
    StkId adr_ptr = L->top - 1;
    StkId adr_ptr_idx = L->ci->func + idx;
    assert(adr_ptr_idx <= adr_ptr); // 注意这里是<=, 也就是说只能在top下方取
    return L->ci->func + idx;
  } else {
    assert(adr_ptr + idx >= L->ci->func);
    return L->top + idx;
  }
}
~~~

#### 在lua栈中, 有savestack(), restorestack(), 请谈一谈你对它们的理解
- 俩函数跟下面三个有关系
	- 首地址
	- 偏移
	- ptr
- 变化的永远是首地址, 偏移永远不变, pr相对首地址变化
- savestack()是为了求取偏移
- restorestack()则是根据新首地址以及偏移更新ptr

#### 我们现在想实现lua_growstack()函数, 然后呢, 假设所有情况都满足, 扩容需求是LUA_STACKSIZE的两倍, 请实现一下lua_growstack()函数
~~~C
#define savestack(L, ptr) ((ptr) - L->stack)
#define restorestack(L, diff) (L->stack + diff)

void luaD_growstack(lua_State* L, int size){

  int size = 2 * LUA_STACKSIZE;

  int top_diff = save(L, L->top);
  TValue* old_stack = L->stack;
  L->stack = luaM_realloc(L, L->stack, L->stack_size, sizeof(TValue) * size);
  L->stack_last = L->stack + size;
  L->stack_size = size;
  L->top = restorestack(L, global_top_diff);

  CallInfo* ci;
  ci = &L->base_ci;
  while(ci){
    int diff_func = ci->func - L->stack;
    int diff_top = ci->top - L->stack;
    restorestack(L, diff_func);
    restorestack(L, diff_top);

    ci = ci->next;
  }
}
~~~

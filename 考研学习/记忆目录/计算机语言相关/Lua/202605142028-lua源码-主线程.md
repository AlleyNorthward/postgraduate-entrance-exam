# Anki
#### 请自行设计lua主线程相关的结构体对象. 提示最终需要用LG表示
~~~C
typedef struct lua_State lua_State;

typedef struct global_State {

} global_State;

typedef struct CallInfo {

} CallInfo;

struct lua_State {
  CallInfo base_ci;
};

typedef struct LX {
  lua_State l;
} LX;

typedef struct LG {
  global_State g;
  LX lx;
} LG;
~~~
- 这是示例代码, 按照自己的思想来写就行

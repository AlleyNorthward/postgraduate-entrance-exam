<%*
const types = ["方法", "教材", "真题", "模拟题"];
const type = await tp.system.suggester(types, types);

const majorSubjects = ["数学", "408", "英语/政治", "其它"];
const major = await tp.system.suggester(majorSubjects, majorSubjects);

let subSubjects = [];
let others = "";
let subject = "";

switch(major){
    case "数学":
        subSubjects = ["高等数学", "线性代数", "概率论", "基础"];
        break;
    case "408":
        subSubjects = ["计算机网络", "操作系统", "数据结构", "计算机组成原理"];
        break;
    case "英语/政治":
        subSubjects = ["英语", "政治"];
        break;
    case "其它":
	    others = await tp.system.prompt("课程名");
	    break;
}

if(subSubjects.length === 0){
	subject = others;
} else {
	subject = await tp.system.suggester(subSubjects, subSubjects);
	subject = (subject === "基础") ? "数学" : subject;
}
const title = await tp.system.prompt("文件名");

const id = tp.date.now("YYYYMMDDHHmm");

tR += `---
id: ${id}
type: ${type}
subject: ${subject}
source: ""
page: ""
tags: []
---

|组成|描述|方法|实例|题目|
|:--:|:--:|:--:|:--:|:--:|
`;

await tp.file.rename(`${tp.date.now("YYYYMMDDHHmm")}-${title}`);
%>
